server {
	listen 8080
	host 127.0.0.1
	server_name SimpleWebServer
	error_page 404 /error/404.html
	client_max_body_size 100000
	web_root tools
	index index.html

	location /tools/ {
		dir_route /Users/mmakarov/Documents/webservIntra/tools/
		autoindex off
		max_route_body_size 1000000
		allow POST
    	allow GET
    	allow DELETE
	}
}
