server {
	listen 8080
	host 127.0.0.1
	server_name SimpleWebServer
	error_page 404 /error/404.html
	client_max_body_size 100000000
	web_root tools
	index index.html

	location / {
		default_file home.html
		autoindex off
    	allow GET
		allow POST
		allow DELETE
	}

	location /image-bin/ {
		autoindex on
		allow GET
		allow POST
		allow DELETE
	}

	location /conf/ {
		deny all
		return 418
	}
}
