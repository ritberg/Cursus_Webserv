server {
	listen 8180
	listen 8181
	host 127.0.0.1
	server_name SimpleWebServer
	error_page 404 /404.html
	client_max_body_size 1000000
	web_root tools
	index index.html

	location / {
		dir_route tools
		autoindex off
		allow POST
    	allow GET
    	allow DELETE
	}

	location /cgi-bin/ {
		default_file cgi.php
		autoindex off
		dir_route tools/cgi-bin
		allow POST
    	allow GET
    	allow DELETE
	}

	location /image-bin/ {
		autoindex on
	}

	location /test {
		redirect /test1
	}

	location /test1 {
		redirect /test2
	}

	location /test2 {
		redirect /test3
	}

	location /test3 {
		redirect /image-bin
	}

}
